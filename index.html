# Command to update a specific ID.

-OnSelect

Patch(
    VnbstPlanList,
    LookUp(VnbstPlanList, ID = foundID),
    {
        StartDate: DataCardStartDate_1.SelectedDate,
        EndDate: DataCardEndDate_1.SelectedDate,
        Days: (DataCardEndDate_1.SelectedDate - DataCardStartDate_1.SelectedDate) + 1,
        WorkPlace: DataCardWorkPlace_1.Value,
        WorkContent: DataCardWorkContent_1.Value,
        Persons: DataCardPersons_1.Value,
        RingiNo: DataCardRingiNo_1.Value,
        RegisterDate: Today(),
        YearMonth: Text(DataCardStartDate_1.SelectedDate,"[$-ja-JP]Yyyyym")
    }
);
Notify("変更保存が完了しました。",NotificationType.Success);

Set()	：グローバル変数の定義
UpdateContext()	：コンテキスト変数の定義（画面内のみ）
If()	：条件分岐
Switch()	：複数条件分岐
Concat()	：文字列連結
LookUp()	：データの検索
Patch()	：データソースにレコードを追加・更新
Navigate()	：画面遷移
